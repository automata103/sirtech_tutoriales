// Movimiento del gusano
function moveWorm() {
    let head = points[0];

    if (entering) {
        head.x += 8;
        if (head.x > canvas.width / 2 - 150) entering = false;
    } else {
        const targetX = mouse.x;
        const targetY = mouse.y;

        const loginRect = document.querySelector('.login-container').getBoundingClientRect();
        const margin = 40;

        const nearLogin = head.x > loginRect.left - margin &&
                         head.x < loginRect.right + margin &&
                         head.y > loginRect.top - margin &&
                         head.y < loginRect.bottom + margin;

        if (nearLogin) {
            const centerX = loginRect.left + loginRect.width / 2;
            const centerY = loginRect.top + loginRect.height / 2;

            const dx = head.x - centerX;
            const dy = head.y - centerY;
            const distance = Math.sqrt(dx * dx + dy * dy);

            if (distance > 0) {
                const force = 0.5;
                headAcceleration.x = (dx / distance) * force;
                headAcceleration.y = (dy / distance) * force;
                setEmotion("curious", 0.6);
            }
        } else {
            if (blockingPassword) {
                const box = document.getElementById("password").getBoundingClientRect();
                headAcceleration.x = (box.left + box.width / 2 - head.x) * 0.001;
                headAcceleration.y = (box.top - 25 - head.y) * 0.001;
            } else if (lookingAtEmail) {
                const box = document.getElementById("email").getBoundingClientRect();
                headAcceleration.x = (box.left + box.width / 2 - head.x) * 0.0008;
                headAcceleration.y = (box.top - 25 - head.y) * 0.0008;
            } else {
                headAcceleration.x = (targetX - head.x) * 0.0008;
                headAcceleration.y = (targetY - head.y) * 0.0008;
            }
        }

        headAcceleration.x -= headVelocity.x * 0.03;
        headAcceleration.y -= headVelocity.y * 0.03;

        headVelocity.x += headAcceleration.x;
        headVelocity.y += headAcceleration.y;

        head.x += headVelocity.x;
        head.y += headVelocity.y;

        const speed = Math.sqrt(headVelocity.x * headVelocity.x + headVelocity.y * headVelocity.y);
        const maxSpeed = 15;
        if (speed > maxSpeed) {
            headVelocity.x = (headVelocity.x / speed) * maxSpeed;
            headVelocity.y = (headVelocity.y / speed) * maxSpeed;
        }
    }

    for (let i = 1; i < segments; i++) {
        let prev = points[i - 1];
        let curr = points[i];
        let dx = prev.x - curr.x;
        let dy = prev.y - curr.y;
        let dist = Math.sqrt(dx * dx + dy * dy);
        if (dist > segmentDistance) {
            let force = Math.min(10, (dist - segmentDistance) * 0.7);
            curr.x += (dx / dist) * force;
            curr.y += (dy / dist) * force;
        }
    }
}

// Dibujar el gusano
function drawWorm() {
    const head = points[0];

    for (let i = segments - 1; i >= 1; i--) {
        const size = 16 - i * 0.3;
        const opacity = 1 - (i / segments) * 0.7;

        ctx.fillStyle = `rgba(${hexToRgb(glowColor)},${opacity * 0.7})`;
        ctx.beginPath();
        ctx.arc(points[i].x, points[i].y, size, 0, Math.PI * 2);
        ctx.fill();

        const gradient = ctx.createRadialGradient(
            points[i].x, points[i].y, 0,
            points[i].x, points[i].y, size
        );
        gradient.addColorStop(0, wormColor);
        gradient.addColorStop(1, adjustColor(wormColor, -50));

        ctx.fillStyle = gradient;
        ctx.beginPath();
        ctx.arc(points[i].x, points[i].y, size * 0.8, 0, Math.PI * 2);
        ctx.fill();
    }

    const headGradient = ctx.createRadialGradient(
        head.x, head.y, 0,
        head.x, head.y, 25
    );
    headGradient.addColorStop(0, wormColor);
    headGradient.addColorStop(1, adjustColor(wormColor, -30));

    ctx.fillStyle = headGradient;
    ctx.beginPath();
    ctx.arc(head.x, head.y, 20, 0, Math.PI * 2);
    ctx.fill();

    ctx.fillStyle = `rgba(${hexToRgb(glowColor)},0.4)`;
    ctx.beginPath();
    ctx.arc(head.x, head.y, 25, 0, Math.PI * 2);
    ctx.fill();

    drawEyes(head);
    drawMouth(head);
}

// Dibujar ojos
function drawEyes(head) {
    const eyeOffsetX = 10;
    const eyeOffsetY = 6;
    let targetX = mouse.x;
    let targetY = mouse.y;

    if (emotion === "frustrated") {
        targetX += (Math.random() - 0.5) * 10;
        targetY += (Math.random() - 0.5) * 10;
    } else if (emotion === "sleepy") {
        targetX = head.x;
        targetY = head.y + 10;
    }

    const angle = Math.atan2(targetY - head.y, targetX - head.x);
    const eyeMoveX = Math.cos(angle) * 4;
    const eyeMoveY = Math.sin(angle) * 4;

    if (!wearingGlasses) {
        let eyeSize = emotion === "alert" ? 7 : 6;
        let pupilSize = emotion === "alert" ? 2.5 : 3;

        if (emotion === "sleepy") {
            eyeSize = 5;
            pupilSize = 2;
        }

        ctx.fillStyle = eyeColor;
        ctx.beginPath();
        ctx.arc(head.x - eyeOffsetX, head.y - eyeOffsetY, eyeSize, 0, Math.PI * 2);
        ctx.fill();
        ctx.fillStyle = pupilColor;
        ctx.beginPath();
        ctx.arc(head.x - eyeOffsetX + eyeMoveX, head.y - eyeOffsetY + eyeMoveY, pupilSize, 0, Math.PI * 2);
        ctx.fill();

        ctx.fillStyle = eyeColor;
        ctx.beginPath();
        ctx.arc(head.x + eyeOffsetX, head.y - eyeOffsetY, eyeSize, 0, Math.PI * 2);
        ctx.fill();
        ctx.fillStyle = pupilColor;
        ctx.beginPath();
        ctx.arc(head.x + eyeOffsetX + eyeMoveX, head.y - eyeOffsetY + eyeMoveY, pupilSize, 0, Math.PI * 2);
        ctx.fill();

        if (emotion === "sleepy") {
            ctx.fillStyle = adjustColor(wormColor, -20);
            ctx.beginPath();
            ctx.arc(head.x - eyeOffsetX, head.y - eyeOffsetY - 2, eyeSize - 1, 0, Math.PI, false);
            ctx.fill();

            ctx.beginPath();
            ctx.arc(head.x + eyeOffsetX, head.y - eyeOffsetY - 2, eyeSize - 1, 0, Math.PI, false);
            ctx.fill();
        }
    }
}

// Dibujar boca
function drawMouth(head) {
    ctx.strokeStyle = pupilColor;
    ctx.lineWidth = 2;

    switch(emotion) {
        case "happy":
            ctx.beginPath();
            ctx.arc(head.x, head.y + 8, 10, 0.2, Math.PI - 0.2, false);
            ctx.stroke();
            break;
        case "frustrated":
            ctx.beginPath();
            ctx.arc(head.x, head.y + 12, 8, 0.1, Math.PI - 0.1, true);
            ctx.stroke();
            break;
        case "sleepy":
            ctx.beginPath();
            ctx.arc(head.x, head.y + 10, 6, 0, Math.PI, false);
            ctx.stroke();
            break;
        default:
            ctx.beginPath();
            ctx.moveTo(head.x - 8, head.y + 8);
            ctx.lineTo(head.x + 8, head.y + 8);
            ctx.stroke();
    }
}

// Sistema de emociones
function setEmotion(newEmotion, intensity = 1.0) {
    targetEmotion = newEmotion;
    targetIntensity = intensity;
    emotionTransition = 0;
    updateEmotionDisplay();
}

function updateEmotion() {
    if (emotion !== targetEmotion || emotionIntensity !== targetIntensity) {
        emotionTransition += 0.05;
        if (emotionTransition >= 1) {
            emotion = targetEmotion;
            emotionIntensity = targetIntensity;
        } else {
            emotionIntensity = emotionIntensity + (targetIntensity - emotionIntensity) * 0.1;
        }
        updateColorsByEmotion();
    }

    const now = Date.now();
    if (now - lastInteraction > 10000 && !sleeping && emotion === "neutral") {
        sleeping = true;
        sleepTimer = 0;
    }

    if (sleeping) {
        sleepTimer += 0.02;
        if (sleepTimer > Math.PI * 2) sleepTimer = 0;
    }
}

function updateColorsByEmotion() {
    switch(emotion) {
        case "alert":
            wormColor = "#ff4f6a";
            glowColor = "rgba(255,70,100,0.8)";

function updateParticles() {
    let now = performance.now();
    particles = particles.filter(p => {
        let age = now - p.born;
        if (age > p.life) return false;
        p.x += p.vx;
        p.y += p.vy;
        p.vy += 0.05;
        p.opacity = 1 - age / p.life;
        return true;
    });
}

function hexToRgb(hex) {
    const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
    return result ?
        `${parseInt(result[1], 16)}, ${parseInt(result[2], 16)}, ${parseInt(result[3], 16)}` :
        "0, 255, 200";
}

function adjustColor(hex, amount) {
    const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
    if (!result) return hex;

    let r = parseInt(result[1], 16) + amount;
    let g = parseInt(result[2], 16) + amount;
    let b = parseInt(result[3], 16) + amount;

    r = Math.min(255, Math.max(0, r));
    g = Math.min(255, Math.max(0, g));
    b = Math.min(255, Math.max(0, b));

    return `#${Math.round(r).toString(16).padStart(2, '0')}${Math.round(g).toString(16).padStart(2, '0')}${Math.round(b).toString(16).padStart(2, '0')}`;
}
